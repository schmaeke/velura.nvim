\newcommand{\periodicElement}[6]{%
	\draw[#5, fill = #6, line width = 0.5mm] (-1.0cm, -1.0cm) rectangle ++(2.0cm, 2.0cm);
	\node[#5] at (0cm, 0cm) {\Huge#2};
	\node[#5, above] at (0cm, -1.0cm) {\sffamily\small#3};
	\node[#5, below right] at (-1.0cm, 1.0cm) {\sffamily\small\bfseries#1};
	\node[#5, below left] at (1.0cm, 1.0cm) {\sffamily\small#4};
}%

	\node[fg, inner sep = 0mm] at (electron.north east) {\Huge -};%
}%
